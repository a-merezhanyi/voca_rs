pub fn is_uppercase(subject: &str) -> bool

Checks whether subject has only upper case characters.

Arguments

  • subject - The string to verify.

Example

use voca_rs::*;
query::is_uppercase("ACDC");
// => true
query::is_uppercase("Morning");
// => false
use voca_rs::Voca;
"ACDC"._is_uppercase();
// => true
